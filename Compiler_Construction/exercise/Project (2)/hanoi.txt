	go
		imagine (n-->1){
			throw_up(^Sposta il disco da ^ + origin + ^ a ^ + destination);
		}nah{
			hanoi(n-1, origin, temp, destination);
			throw_up(^Sposta il disco da ^ + origin + ^ a ^ + destination);
			hanoi(n-1, temp, destination, origin);
		}
	all_set;

}
